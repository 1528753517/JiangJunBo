#include<iostream>
#include<string>
using namespace std;
class student
{
private:
	string name;
	int age;
public:
	friend ostream &operator<<(ostream&, student &a);
	friend istream&operator >> (istream&, student &a);

};
istream&operator >> (istream&input, student &a)
{
	cout << "请输入姓名和年龄"<<endl;
	input >> a.name >> a.age;
	return input;
}
ostream &operator<<(ostream&output, student &a)
{
	output << "name:" << a.name << endl;
	output << "age:" << a.age << endl;
	return output;
}
int main()
{
	student a;
	cin >> a;
	cout << a << endl;
	getchar();
	getchar();

}
