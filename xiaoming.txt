#include <iostream>
using namespace std;
class people      //定义一个people类
{
public:
	char name[10];
};
class student :virtual public people  //从people类虚继承student类
{
public:
	void action1()     //定义构造行数
	{
		cout <<  "小明去上学" << endl;      //输出信息
	}
	void action2()     //定义构造函数
	{
		cout << "小明交作业" << endl;      //输出信息
	}
};
class teacher :virtual public people  //从people类虚继承teacher类
{
public:
	void action1()     //定义构造行数
	{
		cout << "老师布置作业" << endl;      //输出信息
	}
};
int main()
{
	student a;
	teacher b;
	a.action1();
	b.action1();
	a.action2();
	return 0;
}
